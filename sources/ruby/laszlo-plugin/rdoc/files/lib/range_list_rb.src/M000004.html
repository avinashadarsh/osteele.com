  def to_sql_condition
    @ranges.map do |first, last|
      if first == last
        "id=#{first}"
      else
        conjuncts = []
        conjuncts << "#{first}<=id" if first
        conjuncts << "id<=#{last}" if last
        conjuncts.length > 1 ? '(' + conjuncts.join(' AND ') + ')' : conjuncts.first
      end
    end.join(' OR ')
  end
